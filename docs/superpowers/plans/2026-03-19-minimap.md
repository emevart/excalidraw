# Minimap — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Toggleable minimap near zoom controls showing simplified scene overview with viewport rectangle. Click to navigate, drag viewport to pan.

**Architecture:** New `Minimap` React component with its own `<canvas>` element. Rendered inside Footer next to ZoomActions. Uses `useApp()` context hook for scene access. Scene bounds cached via `scene.getSceneNonce()`. Viewport drag converts minimap pixel coords to scene scroll coords. Hidden on phone formFactor and in zen mode.

**Tech Stack:** Canvas 2D API, `getCommonBounds()` from element package, `useApp()` / `useExcalidrawAppState()` hooks, rAF-debounced rendering.

**Spec:** `docs/superpowers/specs/2026-03-18-board-features-design.md` § 4

- [ ] **Step 2: Create Minimap.tsx**

```tsx
import { useCallback, useEffect, useRef, useState } from "react";

import { getCommonBounds } from "@excalidraw/element";
import type { Bounds } from "@excalidraw/common";

import { useApp, useExcalidrawAppState } from "./App";
import { MinimapIcon } from "./icons";
import { t } from "../i18n";
import { Tooltip } from "./Tooltip";

import "./Minimap.scss";

const MINIMAP_WIDTH = 150;
const MINIMAP_HEIGHT = 100;
const PADDING = 10;

type SceneBounds = {
  totalBounds: Bounds;
  scale: number;
  offsetX: number;
  offsetY: number;
};

const computeSceneBounds = (
  elementBounds: Bounds | null,
  vpMinX: number,
  vpMinY: number,
  vpMaxX: number,
  vpMaxY: number,
): SceneBounds | null => {
  const totalMinX = elementBounds ? Math.min(elementBounds[0], vpMinX) : vpMinX;
  const totalMinY = elementBounds ? Math.min(elementBounds[1], vpMinY) : vpMinY;
  const totalMaxX = elementBounds ? Math.max(elementBounds[2], vpMaxX) : vpMaxX;
  const totalMaxY = elementBounds ? Math.max(elementBounds[3], vpMaxY) : vpMaxY;

  const sceneW = totalMaxX - totalMinX;
  const sceneH = totalMaxY - totalMinY;

  if (sceneW === 0 || sceneH === 0) {
    return null;
  }

  const scale = Math.min(
    (MINIMAP_WIDTH - 2 * PADDING) / sceneW,
    (MINIMAP_HEIGHT - 2 * PADDING) / sceneH,
  );

  // Center the content in the minimap
  const offsetX = PADDING + (MINIMAP_WIDTH - 2 * PADDING - sceneW * scale) / 2;
  const offsetY = PADDING + (MINIMAP_HEIGHT - 2 * PADDING - sceneH * scale) / 2;

  return {
    totalBounds: [totalMinX, totalMinY, totalMaxX, totalMaxY],
    scale,
    offsetX,
    offsetY,
  };
};

export const Minimap = () => {
  const app = useApp();
  const appState = useExcalidrawAppState();
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const boundsRef = useRef<SceneBounds | null>(null);
  const rafRef = useRef<number | null>(null);
  const isDraggingRef = useRef(false);

  const toMinimapX = useCallback((sceneX: number) => {
    const b = boundsRef.current;
    if (!b) return 0;
    return (sceneX - b.totalBounds[0]) * b.scale + b.offsetX;
  }, []);

  const toMinimapY = useCallback((sceneY: number) => {
    const b = boundsRef.current;
    if (!b) return 0;
    return (sceneY - b.totalBounds[1]) * b.scale + b.offsetY;
  }, []);

  const toSceneX = useCallback((minimapX: number) => {
    const b = boundsRef.current;
    if (!b) return 0;
    return (minimapX - b.offsetX) / b.scale + b.totalBounds[0];
  }, []);

  const toSceneY = useCallback((minimapY: number) => {
    const b = boundsRef.current;
    if (!b) return 0;
    return (minimapY - b.offsetY) / b.scale + b.totalBounds[1];
  }, []);

  // Render minimap
  useEffect(() => {
    if (!appState.showMinimap) {
      return;
    }

    if (rafRef.current !== null) {
      cancelAnimationFrame(rafRef.current);
    }

    rafRef.current = requestAnimationFrame(() => {
      rafRef.current = null;
      const canvas = canvasRef.current;
      if (!canvas) return;

      const ctx = canvas.getContext("2d");
      if (!ctx) return;

      const dpr = window.devicePixelRatio || 1;
      canvas.width = MINIMAP_WIDTH * dpr;
      canvas.height = MINIMAP_HEIGHT * dpr;
      ctx.scale(dpr, dpr);

      const elements = app.scene.getNonDeletedElements();

      // Viewport in scene coords
      const vpMinX = -appState.scrollX;
      const vpMinY = -appState.scrollY;
      const vpMaxX = vpMinX + appState.width / appState.zoom.value;
      const vpMaxY = vpMinY + appState.height / appState.zoom.value;

      const elBounds = elements.length > 0 ? getCommonBounds(elements) : null;

      const bounds = computeSceneBounds(
        elBounds,
        vpMinX,
        vpMinY,
        vpMaxX,
        vpMaxY,
      );
      boundsRef.current = bounds;

      // Background
      ctx.fillStyle = appState.viewBackgroundColor || "var(--island-bg-color)";
      ctx.fillRect(0, 0, MINIMAP_WIDTH, MINIMAP_HEIGHT);

      if (!bounds) return;

      const { scale } = bounds;

      // Draw elements as simplified rectangles
      const elementsMap = app.scene.getNonDeletedElementsMap();
      for (const el of elements) {
        if (el.isDeleted) continue;

        const [elMinX, elMinY, elMaxX, elMaxY] = getCommonBounds(
          [el],
          elementsMap,
        );
        const w = (elMaxX - elMinX) * scale;
        const h = (elMaxY - elMinY) * scale;

        // Skip elements too small to see
        if (w < 1 && h < 1) continue;

        ctx.fillStyle =
          appState.theme === "dark"
            ? "rgba(255, 255, 255, 0.3)"
            : "rgba(0, 0, 0, 0.25)";
        ctx.fillRect(
          toMinimapX(elMinX),
          toMinimapY(elMinY),
          Math.max(w, 1),
          Math.max(h, 1),
        );
      }

      // Draw viewport rectangle
      ctx.strokeStyle = "rgba(59, 130, 246, 0.8)";
      ctx.fillStyle = "rgba(59, 130, 246, 0.08)";
      ctx.lineWidth = 1.5;
      const vx = toMinimapX(vpMinX);
      const vy = toMinimapY(vpMinY);
      const vw = (vpMaxX - vpMinX) * scale;
      const vh = (vpMaxY - vpMinY) * scale;
      ctx.fillRect(vx, vy, vw, vh);
      ctx.strokeRect(vx, vy, vw, vh);
    });

    return () => {
      if (rafRef.current !== null) {
        cancelAnimationFrame(rafRef.current);
        rafRef.current = null;
      }
    };
  }, [
    app,
    appState.showMinimap,
    appState.scrollX,
    appState.scrollY,
    appState.zoom,
    appState.width,
    appState.height,
    appState.viewBackgroundColor,
    appState.theme,
    // Re-render when scene changes
    app.scene.getSceneNonce(),
    toMinimapX,
    toMinimapY,
  ]);

  // Navigate on click / drag
  const navigateToMinimapPoint = useCallback(
    (clientX: number, clientY: number) => {
      const canvas = canvasRef.current;
      if (!canvas || !boundsRef.current) return;

      const rect = canvas.getBoundingClientRect();
      const mx = clientX - rect.left;
      const my = clientY - rect.top;

      const sceneX = toSceneX(mx);
      const sceneY = toSceneY(my);

      // Center viewport on this scene point
      const newScrollX = -(sceneX - appState.width / (2 * appState.zoom.value));
      const newScrollY = -(
        sceneY -
        appState.height / (2 * appState.zoom.value)
      );

      app.updateScene({
        appState: { scrollX: newScrollX, scrollY: newScrollY },
      });
    },
    [app, appState.width, appState.height, appState.zoom, toSceneX, toSceneY],
  );

  const handlePointerDown = useCallback(
    (e: React.PointerEvent<HTMLCanvasElement>) => {
      isDraggingRef.current = true;
      e.currentTarget.setPointerCapture(e.pointerId);
      navigateToMinimapPoint(e.clientX, e.clientY);
    },
    [navigateToMinimapPoint],
  );

  const handlePointerMove = useCallback(
    (e: React.PointerEvent<HTMLCanvasElement>) => {
      if (!isDraggingRef.current) return;
      navigateToMinimapPoint(e.clientX, e.clientY);
    },
    [navigateToMinimapPoint],
  );

  const handlePointerUp = useCallback(() => {
    isDraggingRef.current = false;
  }, []);

  if (!appState.showMinimap) {
    return null;
  }

  return (
    <div className="minimap__canvas-container">
      <canvas
        ref={canvasRef}
        className="minimap__canvas"
        style={{ width: MINIMAP_WIDTH, height: MINIMAP_HEIGHT }}
        onPointerDown={handlePointerDown}
        onPointerMove={handlePointerMove}
        onPointerUp={handlePointerUp}
      />
    </div>
  );
};

export const MinimapToggle = () => {
  const appState = useExcalidrawAppState();
  const app = useApp();

  const toggle = useCallback(() => {
    app.updateScene({
      appState: { showMinimap: !appState.showMinimap },
    });
  }, [app, appState.showMinimap]);

  return (
    <div className="minimap">
      <Tooltip label={t("labels.minimap")}>
        <button
          className={`minimap__toggle ${
            appState.showMinimap ? "minimap__toggle--active" : ""
          }`}
          onClick={toggle}
          aria-label={t("labels.minimap")}
        >
          {MinimapIcon}
        </button>
      </Tooltip>
      <Minimap />
    </div>
  );
};
```
